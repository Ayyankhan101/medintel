# Plan 08: Medical Resource Tracking

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Answer "Where is the nearest oxygen cylinder?", "Where is the nearest ventilator?", "Which pharmacy has this specific injection in stock?" — displayed on a map with one-click directions.

**Architecture:** Resource types extend the existing `MedicalResource` table (OXYGEN, VENTILATOR, PHARMACY, HOSPITAL). Inventory is tracked in a separate `ResourceInventory` table for real-time stock. The `/resources` page lets users search by type + their GPS location.

**Tech Stack:** Prisma, Google Maps, Next.js

**Depends on:** Plan 07 (MedicalResource table + nearby API)

model ResourceInventory {

### Task 2: Resource Search API

**Files:**
- Create: `app/api/resources/route.ts`

- [ ] **Step 1: Create resource search endpoint**

Create `app/api/resources/route.ts`:
```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

export async function GET(req: NextRequest) {
  const { searchParams } = new URL(req.url)
  const type = searchParams.get('type')
  const query = searchParams.get('q')        // item name search e.g. "oxygen", "ventilator"

  const resources = await prisma.medicalResource.findMany({
    where: {
      isAvailable: true,
      ...(type ? { type } : {}),
      ...(query ? { name: { contains: query, mode: 'insensitive' } } : {}),
    },
    include: {
      inventory: {
        where: query
          ? { itemName: { contains: query, mode: 'insensitive' }, quantity: { gt: 0 } }
          : { quantity: { gt: 0 } },
      },
    },
    orderBy: { name: 'asc' },
    take: 20,
  })

  // Filter to only resources with relevant inventory if a query was given
  const filtered = query
    ? resources.filter(r => r.inventory.length > 0 || r.type === type)
    : resources

  return NextResponse.json(filtered)
}
```

### Task 4: Resource Search Page

**Files:**
- Create: `app/(patient)/resources/page.tsx`

- [ ] **Step 1: Create resource search UI**

Create `app/(patient)/resources/page.tsx`:
```tsx
'use client'
import { useState, useEffect } from 'react'
import { Input } from '@/components/ui/input'
import { Button } from '@/components/ui/button'
import { Badge } from '@/components/ui/badge'
import { MapPin, Phone, Package } from 'lucide-react'

const RESOURCE_TYPES = [
  { value: '', label: 'All' },
  { value: 'HOSPITAL', label: 'Hospitals' },
  { value: 'CATH_LAB', label: 'Cath Labs' },
  { value: 'PHARMACY', label: 'Pharmacies' },
  { value: 'OXYGEN', label: 'Oxygen' },
  { value: 'VENTILATOR', label: 'Ventilators' },
]

const typeColor: Record<string, string> = {
  HOSPITAL:   'bg-blue-100 text-blue-700',
  CATH_LAB:   'bg-red-100 text-red-700',
  PHARMACY:   'bg-green-100 text-green-700',
  OXYGEN:     'bg-sky-100 text-sky-700',
  VENTILATOR: 'bg-purple-100 text-purple-700',
}

export default function ResourcesPage() {
  const [query, setQuery] = useState('')
  const [type, setType] = useState('')
  const [resources, setResources] = useState<any[]>([])
  const [loading, setLoading] = useState(false)
  const [userLat, setUserLat] = useState<number | null>(null)
  const [userLng, setUserLng] = useState<number | null>(null)

  useEffect(() => {
    navigator.geolocation.getCurrentPosition(
      pos => { setUserLat(pos.coords.latitude); setUserLng(pos.coords.longitude) },
      () => {}
    )
    // Load initial list
    search()
  }, [])

  async function search() {
    setLoading(true)
    const params = new URLSearchParams()
    if (type) params.set('type', type)
    if (query) params.set('q', query)
    const res = await fetch(`/api/resources?${params}`)
    setResources(await res.json())
    setLoading(false)
  }

  function getDirectionsUrl(lat: number, lng: number) {
    const origin = userLat && userLng ? `${userLat},${userLng}` : ''
    return `https://www.google.com/maps/dir/${origin}/${lat},${lng}`
  }

  return (
    <div className="max-w-2xl mx-auto p-4 space-y-4">
      <h1 className="text-2xl font-bold">Find Medical Resources</h1>

      <div className="flex gap-2">
        <Input
          value={query}
          onChange={e => setQuery(e.target.value)}
          placeholder="Search: oxygen cylinder, ventilator, Augmentin..."
          onKeyDown={e => e.key === 'Enter' && search()}
          className="flex-1"
        />
        <Button onClick={search} disabled={loading}>Search</Button>
      </div>

      <div className="flex gap-2 flex-wrap">
        {RESOURCE_TYPES.map(t => (
          <button
            key={t.value}
            onClick={() => { setType(t.value); }}
            className={`px-3 py-1 rounded-full text-sm border transition-colors ${type === t.value ? 'bg-gray-900 text-white border-gray-900' : 'border-gray-300 text-gray-600 hover:border-gray-500'}`}
          >
            {t.label}
          </button>
        ))}
      </div>

      {loading ? (
        <div className="space-y-3">{Array(4).fill(0).map((_, i) => (
          <div key={i} className="h-20 bg-gray-100 rounded-xl animate-pulse" />
        ))}</div>
      ) : (
        <div className="space-y-3">
          {resources.map(r => (
            <div key={r.id} className="border rounded-xl p-4 bg-white space-y-2">
              <div className="flex items-start justify-between gap-2">
                <div>
                  <div className="flex items-center gap-2">
                    <span className="font-semibold text-gray-900">{r.name}</span>
                    <Badge className={typeColor[r.type] ?? 'bg-gray-100 text-gray-700'}>{r.type}</Badge>
                  </div>
                  <p className="text-sm text-gray-500 flex items-center gap-1 mt-0.5">
                    <MapPin className="w-3 h-3" /> {r.address}
                  </p>
                </div>
                <a
                  href={getDirectionsUrl(Number(r.latitude), Number(r.longitude))}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="text-sm text-blue-600 hover:underline shrink-0"
                >
                  Directions
                </a>
              </div>

              {r.inventory && r.inventory.length > 0 && (
                <div className="flex gap-2 flex-wrap">
                  {r.inventory.map((inv: any) => (
                    <span key={inv.id} className="inline-flex items-center gap-1 text-xs bg-green-50 text-green-700 border border-green-200 rounded px-2 py-0.5">
                      <Package className="w-3 h-3" /> {inv.itemName}: <strong>{inv.quantity} in stock</strong>
                    </span>
                  ))}
                </div>
              )}

              {r.phone && (
                <a href={`tel:${r.phone}`} className="inline-flex items-center gap-1 text-sm text-blue-600 hover:underline">
                  <Phone className="w-3 h-3" /> {r.phone}
                </a>
              )}
            </div>
          ))}
          {resources.length === 0 && !loading && (
            <p className="text-center text-gray-400 py-8">No resources found. Try a different search.</p>
          )}
        </div>
      )}
    </div>
  )
}
```
