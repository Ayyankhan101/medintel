# Plan 07: Emergency Protocol + Geo-Mapping

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** When triage detects a severity score ≥ 8 (e.g., chest pain + high troponin), immediately: (1) play audio first-aid guidance, (2) show the 3 nearest hospitals with emergency facilities on a map, (3) highlight top-rated interventional cardiologists.

**Architecture:** Emergency detection is pure logic in `lib/emergency.ts`. Audio guidance is pre-recorded MP3s served from S3. Geo-mapping uses Google Maps JS API. Hospital data lives in the `MedicalResource` table seeded with Pakistani hospital data.

**Tech Stack:** Google Maps JS API, Web Audio API, Prisma, Next.js API routes, Vitest

**Depends on:** Plan 01 (MedicalResource table), Plan 03 (triage severity score)

import { NextRequest, NextResponse } from 'next/server'
import { z } from 'zod'
import { auth } from '@/lib/auth'
import { prisma } from '@/lib/prisma'
import { detectEmergencyCondition, getFirstAidGuidance } from '@/lib/emergency'
import { getPresignedDownloadUrl } from '@/lib/s3'

const schema = z.object({ appointmentId: z.string().optional(), text: z.string().optional() })

export async function POST(req: NextRequest) {
  const session = await auth()
  if (!session?.user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const body = await req.json()
  const parsed = schema.safeParse(body)
  if (!parsed.success) return NextResponse.json({ error: parsed.error.flatten() }, { status: 400 })

  let textToAnalyze = parsed.data.text ?? ''

  if (parsed.data.appointmentId) {
    const appointment = await prisma.appointment.findUnique({
      where: { id: parsed.data.appointmentId },
    })
    textToAnalyze += ` ${appointment?.aiSummary ?? ''} ${appointment?.transcript ?? ''}`
  }

  const emergency = detectEmergencyCondition(textToAnalyze)
  if (!emergency) {
    return NextResponse.json({ isEmergency: false })
  }

  const guidance = getFirstAidGuidance(emergency.condition)
  const audioUrl = await getPresignedDownloadUrl(guidance.audioKey, 3600)

  return NextResponse.json({
    isEmergency: true,
    condition: emergency.condition,
    requiredFacility: emergency.requiredFacility,
    requiredSpecialty: emergency.requiredSpecialty,
    guidance: {
      text: guidance.text,
      steps: guidance.steps,
      audioUrl,
    },
  })
}

import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

// Haversine formula — distance in km between two lat/lng points
function haversineKm(lat1: number, lng1: number, lat2: number, lng2: number): number {
  const R = 6371
  const dLat = ((lat2 - lat1) * Math.PI) / 180
  const dLng = ((lng2 - lng1) * Math.PI) / 180
  const a =
    Math.sin(dLat / 2) ** 2 +
    Math.cos((lat1 * Math.PI) / 180) * Math.cos((lat2 * Math.PI) / 180) * Math.sin(dLng / 2) ** 2
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
}

export async function GET(req: NextRequest) {
  const { searchParams } = new URL(req.url)
  const lat = parseFloat(searchParams.get('lat') ?? '0')
  const lng = parseFloat(searchParams.get('lng') ?? '0')
  const type = searchParams.get('type')        // CATH_LAB, HOSPITAL, PHARMACY, OXYGEN, VENTILATOR
  const limitStr = searchParams.get('limit') ?? '3'
  const limit = Math.min(parseInt(limitStr), 10)

  if (!lat || !lng) return NextResponse.json({ error: 'lat and lng are required' }, { status: 400 })

  const resources = await prisma.medicalResource.findMany({
    where: {
      isAvailable: true,
      ...(type ? { type } : {}),
    },
  })

  const withDistance = resources
    .map(r => ({
      ...r,
      distanceKm: haversineKm(lat, lng, Number(r.latitude), Number(r.longitude)),
    }))
    .sort((a, b) => a.distanceKm - b.distanceKm)
    .slice(0, limit)

  return NextResponse.json(withDistance)
}

'use client'
import { useEffect, useRef, useState } from 'react'
import { AlertTriangle, Volume2 } from 'lucide-react'
import { Button } from '@/components/ui/button'

interface Props {
  condition: string
  guidanceText: string
  audioUrl: string
  steps: string[]
}

export function EmergencyAlert({ condition, guidanceText, audioUrl, steps }: Props) {
  const audioRef = useRef<HTMLAudioElement | null>(null)
  const [audioPlaying, setAudioPlaying] = useState(false)

  useEffect(() => {
    // Auto-play guidance audio on mount
    const audio = new Audio(audioUrl)
    audioRef.current = audio
    audio.onended = () => setAudioPlaying(false)
    audio.play().then(() => setAudioPlaying(true)).catch(() => {})
    return () => { audio.pause(); audio.src = '' }
  }, [audioUrl])

  function replayAudio() {
    if (audioRef.current) {
      audioRef.current.currentTime = 0
      audioRef.current.play()
      setAudioPlaying(true)
    }
  }

  return (
    <div className="bg-red-600 text-white rounded-2xl p-6 space-y-4 shadow-xl">
      <div className="flex items-center gap-3">
        <AlertTriangle className="w-8 h-8 animate-pulse shrink-0" />
        <div>
          <h2 className="text-xl font-bold">EMERGENCY DETECTED</h2>
          <p className="text-red-200 text-sm">{condition}</p>
        </div>
        <Button
          variant="ghost"
          size="icon"
          onClick={replayAudio}
          className="ml-auto text-white hover:bg-red-500"
        >
          <Volume2 className={`w-6 h-6 ${audioPlaying ? 'animate-pulse' : ''}`} />
        </Button>
      </div>

        <p className="font-medium leading-relaxed">{guidanceText}</p>
      </div>

'use client'
import { useEffect, useRef } from 'react'

interface Resource {
  id: string
  name: string
  type: string
  address: string
  latitude: number | string
  longitude: number | string
  distanceKm: number
  phone: string | null
}

interface Props {
  resources: Resource[]
  userLat: number
  userLng: number
}

declare global {
  interface Window { google: any; initMap: () => void }
}

export function ResourceMap({ resources, userLat, userLng }: Props) {
  const mapRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    if (!mapRef.current || !window.google) return

    const map = new window.google.maps.Map(mapRef.current, {
      center: { lat: userLat, lng: userLng },
      zoom: 13,
      styles: [{ featureType: 'poi.medical', elementType: 'labels', stylers: [{ visibility: 'on' }] }],
    })

    // User location marker
    new window.google.maps.Marker({
      position: { lat: userLat, lng: userLng },
      map,
      icon: { path: window.google.maps.SymbolPath.CIRCLE, scale: 8, fillColor: '#4285F4', fillOpacity: 1, strokeWeight: 2, strokeColor: '#fff' },
      title: 'Your Location',
    })

    // Hospital/resource markers
    resources.forEach((r, i) => {
      const marker = new window.google.maps.Marker({
        position: { lat: Number(r.latitude), lng: Number(r.longitude) },
        map,
        label: { text: String(i + 1), color: 'white', fontWeight: 'bold' },
        icon: { path: window.google.maps.SymbolPath.BACKWARD_CLOSED_ARROW, scale: 8, fillColor: '#dc2626', fillOpacity: 1, strokeWeight: 2, strokeColor: '#fff' },
        title: r.name,
      })

      const infoWindow = new window.google.maps.InfoWindow({
        content: `<div style="max-width:200px"><strong>${r.name}</strong><br/>${r.address}<br/>${r.distanceKm.toFixed(1)} km away${r.phone ? `<br/><a href="tel:${r.phone}">${r.phone}</a>` : ''}</div>`,
      })
      marker.addListener('click', () => infoWindow.open(map, marker))
    })
  }, [resources, userLat, userLng])

  return <div ref={mapRef} className="w-full h-64 rounded-xl overflow-hidden" />
}

'use client'
import { useEffect, useState, Suspense } from 'react'
import { useSearchParams } from 'next/navigation'
import { EmergencyAlert } from '@/components/emergency/EmergencyAlert'
import { ResourceMap } from '@/components/maps/ResourceMap'
import Script from 'next/script'

interface NearbyResource {
  id: string
  name: string
  type: string
  address: string
  latitude: number
  longitude: number
  distanceKm: number
  phone: string | null
}

function EmergencyContent() {
  const params = useSearchParams()
  const appointmentId = params.get('appointmentId')
  const [emergency, setEmergency] = useState<any>(null)
  const [resources, setResources] = useState<NearbyResource[]>([])
  const [location, setLocation] = useState<{ lat: number; lng: number } | null>(null)
  const [mapsLoaded, setMapsLoaded] = useState(false)

  useEffect(() => {
    // Get GPS location
    navigator.geolocation.getCurrentPosition(
      pos => setLocation({ lat: pos.coords.latitude, lng: pos.coords.longitude }),
      () => setLocation({ lat: 31.5497, lng: 74.3436 }) // default: Lahore
    )
  }, [])

  useEffect(() => {
    if (!appointmentId) return
    fetch('/api/emergency', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ appointmentId }),
    }).then(r => r.json()).then(setEmergency)
  }, [appointmentId])

  useEffect(() => {
    if (!location || !emergency?.requiredFacility) return
    fetch(`/api/resources/nearby?lat=${location.lat}&lng=${location.lng}&type=${emergency.requiredFacility}&limit=3`)
      .then(r => r.json())
      .then(setResources)
  }, [location, emergency])

  if (!emergency?.isEmergency) return null

  return (
    <div className="max-w-2xl mx-auto p-4 space-y-6">
      <EmergencyAlert
        condition={emergency.condition}
        guidanceText={emergency.guidance.text}
        audioUrl={emergency.guidance.audioUrl}
        steps={emergency.guidance.steps}
      />

      <div className="space-y-3">
        <h3 className="font-bold text-gray-800">
          Nearest {emergency.requiredFacility === 'CATH_LAB' ? 'Cath Lab Hospitals' : 'Hospitals'}
        </h3>
        {location && mapsLoaded && (
          <ResourceMap resources={resources} userLat={location.lat} userLng={location.lng} />
        )}
        <div className="space-y-2">
          {resources.map((r, i) => (
            <div key={r.id} className="flex items-start gap-3 p-3 border rounded-lg bg-white">
              <span className="bg-red-600 text-white rounded-full w-6 h-6 flex items-center justify-center text-xs font-bold shrink-0">{i + 1}</span>
              <div className="flex-1 min-w-0">
                <p className="font-medium text-gray-900">{r.name}</p>
                <p className="text-sm text-gray-500">{r.address} — {r.distanceKm.toFixed(1)} km</p>
              </div>
              {r.phone && (
                <a href={`tel:${r.phone}`} className="text-blue-600 text-sm shrink-0">Call</a>
              )}
            </div>
          ))}
        </div>
      </div>
    </div>
  )
}

export default function EmergencyPage() {
  return (
    <>
      <Script
        src={`https://maps.googleapis.com/maps/api/js?key=${process.env.NEXT_PUBLIC_GOOGLE_MAPS_API_KEY}`}
        onLoad={() => {}}
        strategy="afterInteractive"
      />
      <Suspense><EmergencyContent /></Suspense>
    </>
  )
}
